# Cài đặt các dependencies cần thiết cho Cây Gia Phả

# 1. React Organizational Chart
npm install react-organizational-chart

# 2. Material-UI (nếu chưa có)
npm install @mui/material @emotion/react @emotion/styled

# 3. Material-UI Icons
npm install @mui/icons-material

# 4. Axios (nếu chưa có)
npm install axios

# 5. React Router (nếu chưa có)
npm install react-router-dom

# Hoặc cài tất cả cùng lúc:
npm install react-organizational-chart @mui/material @emotion/react @emotion/styled @mui/icons-material axios react-router-dom
